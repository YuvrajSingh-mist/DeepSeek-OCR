\[
\begin{align*}
B_X &= X_{s2} - X_{s1} = N_1 X_1 - N_2 X_2 \\
B_Y &= Y_{s2} - Y_{s1} = N_1 Y_1 - N_2 Y_2 \\
B_Z &= Z_{s2} - Z_{s1} = N_1 Z_1 - N_2 Z_2
\end{align*}
\]

\[
\begin{align*}
N_1 &= \frac{B_X Z_2 - B_Z X_2}{X_1 Z_2 - X_2 Z_1} \\
N_2 &= \frac{B_X Z_1 - B_Z X_1}{X_1 Z_2 - X_2 Z_1}
\end{align*}
\]
